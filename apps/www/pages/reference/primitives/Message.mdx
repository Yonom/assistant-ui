import { PropsTable } from "@/components/docs";

## Anatomy

```tsx
import { MessagePrimitive } from "@assistant-ui/react";

const UserMessage = () => (
  <MessagePrimitive.Root>
    User: <MessagePrimitive.Content />
    
    <BranchPicker />
    <ActionBar />
  </Message.Root>
);

const AssistantMessage = () => (
  <MessagePrimitive.Root>
    Assistant: <MessagePrimitive.Content />
    
    <BranchPicker />
    <ActionBar />
  </Message.Root>
);
```

## API Reference

### Root

Containts all parts of the message.

This primitive renders a `<div>` element unless `asChild` is set.

### InProgress

The indicator to show while an assistant message is being generated.  

The contents of this element will be portaled to `<ContentPartPrimitive.InProgressIndicator />`, which allows the indicator to be displayed at the very end of the message.  

This primitive renders a `<span>` element unless `asChild` is set.
